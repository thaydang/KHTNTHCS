"""
exam_to_docx.py
Converts Markdown exam files to professionally formatted .docx files
Compatible with GitHub Actions for automated document generation
"""

from docx import Document
from docx.shared import Pt, RGBColor, Inches
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.oxml.ns import qn
from docx.oxml import OxmlElement
import re
import os
import sys
from pathlib import Path
from datetime import datetime

class ExamDocumentGenerator:
    """Generate professional exam documents from Markdown"""
    
    def __init__(self, md_file_path, output_dir=None):
        self.md_file = md_file_path
        self.output_dir = output_dir or Path(md_file_path).parent
        self.doc = Document()
        self.setup_styles()
        
    def setup_styles(self):
        """Configure document styles"""
        # Set default font
        style = self.doc.styles['Normal']
        style.font.name = 'Calibri'
        style.font.size = Pt(11)
        
    def add_heading(self, text, level=1):
        """Add formatted heading"""
        heading = self.doc.add_heading(text, level=level)
        
        if level == 1:
            heading.style = 'Heading 1'
            for run in heading.runs:
                run.font.color.rgb = RGBColor(0, 51, 102)
                run.font.bold = True
                run.font.size = Pt(16)
        elif level == 2:
            heading.style = 'Heading 2'
            for run in heading.runs:
                run.font.color.rgb = RGBColor(0, 102, 153)
                run.font.bold = True
                run.font.size = Pt(13)
        
        heading.alignment = WD_ALIGN_PARAGRAPH.LEFT
        return heading
    
    def add_info_table(self, school="", class_num="", student_name=""):
        """Add student information table"""
        table = self.doc.add_table(rows=3, cols=2)
        table.style = 'Light Grid Accent 1'
        
        # Header row
        hdr_cells = table.rows[0].cells
        hdr_cells[0].text = "Nội dung"
        hdr_cells[1].text = "Chi tiết"
        
        # Data rows
        table.rows[1].cells[0].text = "Trường"
        table.rows[1].cells[1].text = school or "................................................................."
        
        table.rows[2].cells[0].text = "Lớp"
        table.rows[2].cells[1].text = class_num or "................................................................."
        
        self.doc.add_paragraph()  # Add spacing
    
    def add_metadata(self, subject="", class_level="", duration="60", scope=""):
        """Add exam metadata"""
        p = self.doc.add_paragraph()
        p.add_run("Môn học: ").bold = True
        p.add_run(subject)
        
        p = self.doc.add_paragraph()
        p.add_run("Khối lớp: ").bold = True
        p.add_run(class_level)
        
        p = self.doc.add_paragraph()
        p.add_run("Thời gian làm bài: ").bold = True
        p.add_run(f"{duration} phút (không kể thời gian phát đề)")
        
        if scope:
            p = self.doc.add_paragraph()
            p.add_run("Phạm vi kiến thức: ").bold = True
            p.add_run(scope)
        
        self.doc.add_paragraph()  # Add spacing
    
    def add_instructions(self, instruction_text):
        """Add general instructions"""
        self.add_heading("HƯỚNG DẪN CHUNG", level=2)
        p = self.doc.add_paragraph(instruction_text)
        p.paragraph_format.line_spacing = 1.5
        self.doc.add_paragraph()
    
    def add_section(self, title, points):
        """Add exam section"""
        self.add_heading(title, level=2)
        p = self.doc.add_paragraph(f"Điểm: {points} điểm")
        p.runs[0].bold = True
        self.doc.add_paragraph()
    
    def add_question(self, question_num, question_text, question_type="multiple_choice"):
        """Add question to document"""
        p = self.doc.add_paragraph()
        run = p.add_run(f"Câu {question_num}: ")
        run.bold = True
        run.font.size = Pt(11)
        p.add_run(question_text)
        
    def add_options(self, options):
        """Add multiple choice options"""
        for option in options:
            p = self.doc.add_paragraph(option, style='List Bullet')
            p.paragraph_format.left_indent = Inches(0.5)
    
    def add_true_false_table(self, items):
        """Add true/false question table"""
        table = self.doc.add_table(rows=len(items) + 1, cols=2)
        table.style = 'Light Grid Accent 1'
        
        # Header
        hdr_cells = table.rows[0].cells
        hdr_cells[0].text = "Nhận định"
        hdr_cells[1].text = "Đúng/Sai"
        
        # Items
        for idx, item in enumerate(items, 1):
            table.rows[idx].cells[0].text = item
            table.rows[idx].cells[1].text = ""
        
        self.doc.add_paragraph()
    
    def add_short_answer(self, question_num, question_text):
        """Add short answer question"""
        p = self.doc.add_paragraph()
        run = p.add_run(f"Câu {question_num}: ")
        run.bold = True
        p.add_run(question_text)
        
        # Add blank lines for answer
        for _ in range(2):
            self.doc.add_paragraph("......................................................................")
        self.doc.add_paragraph()
    
    def add_essay_question(self, question_num, points, level, topic, question_text):
        """Add essay question"""
        p = self.doc.add_paragraph()
        run = p.add_run(f"Câu {question_num} ({points} điểm) - Mức độ {level} ({topic})")
        run.bold = True
        
        p = self.doc.add_paragraph(question_text)
        p.paragraph_format.left_indent = Inches(0.25)
        
        # Add blank lines for answer
        for _ in range(3):
            self.doc.add_paragraph("......................................................................")
        self.doc.add_paragraph()
    
    def add_answer_key(self, section_title, content_dict):
        """Add answer key section"""
        self.add_heading(section_title, level=2)
        
        for key, value in content_dict.items():
            p = self.doc.add_paragraph()
            run = p.add_run(f"{key}: ")
            run.bold = True
            p.add_run(str(value))
    
    def add_page_break(self):
        """Add page break"""
        self.doc.add_page_break()
    
    def save(self, filename=None):
        """Save document"""
        if not filename:
            base_name = Path(self.md_file).stem
            filename = f"{base_name}-Formatted.docx"
        
        output_path = Path(self.output_dir) / filename
        self.doc.save(str(output_path))
        print(f"✓ Document saved: {output_path}")
        return output_path
    
    def parse_and_generate(self):
        """Parse Markdown file and generate document"""
        with open(self.md_file, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # Extract title
        title_match = re.search(r'^#\s+(.+)$', content, re.MULTILINE)
        if title_match:
            self.add_heading(title_match.group(1), level=1)
        
        # Add metadata section
        self.add_heading("THÔNG TIN ĐỀ THI", level=2)
        
        # Extract metadata
        metadata = {}
        for pattern, key in [
            (r'\*\*Môn học:\*\*\s+(.+)', 'Môn học'),
            (r'\*\*Khối lớp:\*\*\s+(.+)', 'Khối lớp'),
            (r'\*\*Thời gian làm bài:\*\*\s+(.+)', 'Thời gian'),
            (r'\*\*Phạm vi kiến thức:\*\*\s+(.+)', 'Phạm vi'),
        ]:
            match = re.search(pattern, content)
            if match:
                metadata[key] = match.group(1)
        
        # Add to document
        for key, value in metadata.items():
            p = self.doc.add_paragraph()
            run = p.add_run(f"{key}: ")
            run.bold = True
            p.add_run(value)
        
        self.doc.add_paragraph()
        
        # Extract and add sections
        sections = re.split(r'^#\s+(.+)$', content, flags=re.MULTILINE)[1:]
        
        for i in range(0, len(sections), 2):
            if i + 1 < len(sections):
                section_title = sections[i]
                section_content = sections[i + 1]
                
                if section_title not in ['ĐỀ KIỂM TRA CUỐI HỌC KÌ I']:
                    self.add_heading(section_title, level=2)
                    
                    # Extract questions
                    questions = re.findall(r'\*\*Câu \d+[^*]+\*\*:\s*(.+?)(?=\*\*Câu|\Z)', 
                                         section_content, re.DOTALL)
                    
                    for question in questions[:5]:  # Limit for demo
                        p = self.doc.add_paragraph(question.strip())
                        p.paragraph_format.left_indent = Inches(0.25)
                    
                    self.doc.add_paragraph()


def batch_convert(directory, output_dir=None):
    """Convert all exam markdown files in directory"""
    directory = Path(directory)
    output_dir = Path(output_dir) if output_dir else directory
    output_dir.mkdir(parents=True, exist_ok=True)
    
    md_files = list(directory.glob("*Kiem-tra*.md")) + list(directory.glob("*De*.md"))
    
    if not md_files:
        print(f"No exam files found in {directory}")
        return
    
    print(f"Found {len(md_files)} exam files to convert")
    
    for md_file in md_files:
        print(f"Processing: {md_file.name}")
        try:
            generator = ExamDocumentGenerator(str(md_file), str(output_dir))
            generator.parse_and_generate()
            generator.save()
        except Exception as e:
            print(f"✗ Error processing {md_file.name}: {e}")


def main():
    """Main function for CLI usage"""
    if len(sys.argv) < 2:
        print("Usage: python exam_to_docx.py <markdown_file> [output_directory]")
        print("       python exam_to_docx.py --batch <directory> [output_directory]")
        sys.exit(1)
    
    if sys.argv[1] == "--batch":
        directory = sys.argv[2] if len(sys.argv) > 2 else "."
        output_dir = sys.argv[3] if len(sys.argv) > 3 else None
        batch_convert(directory, output_dir)
    else:
        md_file = sys.argv[1]
        output_dir = sys.argv[2] if len(sys.argv) > 2 else None
        
        if not Path(md_file).exists():
            print(f"Error: File not found: {md_file}")
            sys.exit(1)
        
        generator = ExamDocumentGenerator(md_file, output_dir)
        generator.parse_and_generate()
        output_path = generator.save()
        print(f"✓ Successfully converted: {output_path}")


if __name__ == "__main__":
    main()
